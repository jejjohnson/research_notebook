def make_checkerboard(n_samples):
    x1 = np.random.rand(n_samples) * 4 - 2
    x2_ = np.random.rand(n_samples) - np.random.randint(0, 2, [n_samples]) * 2
    x2 = x2_ + np.floor(x1) % 2
    data = np.vstack([x1, x2]).T * 2
    return data
